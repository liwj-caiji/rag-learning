    subgraph RULE_PATH["规则路径（QueryRewriter / classify_intent）"]

        R2 --> R3["提取过滤条件"]
        R3 --> R3a["_DIFFICULTY_PATTERNS\n→ '★' / '★★' / '★★★★'"]
        R3a --> R3b["_CATEGORY_KEYWORDS\n→ vegetable_dish / meat_dish / ..."]
        R3b --> R3c["_CALORIE_LOW/HIGH_PATTERNS\n→ 'low' / 'high'"]

        R2 --> R4{"意图是 howto\n或 ingredient?"}
        R4 -->|"是"| R5["_DISH_PATTERN 提取菜名\n去除「的做法」后缀"]

        R2 --> R7["_build_rewritten_query()"]
        R7 --> R7a{"意图?"}
        R7a -->|"recommendation"| R7b["去除语气词\n（今天/吃什么/推荐...）"]
        R7a -->|"howto"| R7c["'{dish_name} 操作 步骤'"]
        R7a -->|"ingredient"| R7d["'{dish_name} 必备原料和工具'"]
        R7a -->|"factual"| R7e["返回原查询"]

        R9a --> R9b["类别探针: '{category} {base}'"]

        R8 -->|"否"| R10["probes = []"]
    end

回退安全网
LLMIntentClassifier 在任何异常（无 API Key、网络超时、JSON 解析失败）下自动降级到基于规则的 classify_intent()，保证系统始终可用。
